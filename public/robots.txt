export function GET() {
  const robots = `User-agent: *
Allow: /
Disallow: /dashboard/
Disallow: /onboarding/
Disallow: /track/
Disallow: /onboardagency/

Sitemap: https://carrydey.tech/sitemap.xml`;

  return new Response(robots, {
    headers: {
      'Content-Type': 'text/plain; charset=utf-8',
      'Cache-Control': 'public, max-age=86400, s-maxage=86400',
    },
  });
}
